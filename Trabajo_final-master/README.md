# Trabajo_final

# Required gems

require 'sinatra'
require 'byebug'
require 'shotgun'
require 'yaml/store'
require 'csv'
require 'date'
require 'pill_chart'


# To install CSV gem

Add this line to application's Gemfile:

gem 'csv'

And then execute:
$ bundle

Or install it yourself as:
$ gem install csv


# To run application:

$ cd 'aplication dir path'
$ shotgun webapp.rb
